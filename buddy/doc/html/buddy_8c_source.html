118 page_t* buddy_combine(page_t* the_page, page_t* buddy_page){
119 
120  // give back original page if its buddy is not free
121  if(!buddy_page->is_free) return the_page;
122 
123  // also if the buddy is not the correct size
124  if(buddy_page->block_size != the_page->block_size) return the_page;

195  while((1 << needed_block) < size) needed_block++;
